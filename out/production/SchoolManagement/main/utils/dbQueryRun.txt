CREATE TABLE `student` (
  `sId` int NOT NULL,
  `sName` varchar(20) NOT NULL,
  PRIMARY KEY (`sId`)
)

CREATE TABLE `face` (
  `fId` int NOT NULL AUTO_INCREMENT,
  `fData` varchar(100) NOT NULL,
  `fSet` int NOT NULL,
  `f_sId` int NOT NULL,
  PRIMARY KEY (`fId`),
  UNIQUE KEY `group_set` (`fSet`,`f_sId`),
  CONSTRAINT `f_sId` FOREIGN KEY (`f_sId`) REFERENCES `student` (`sId`)
)

CREATE TABLE `attendance` (
  `aDate` date NOT NULL,
  `aStatus` boolean NOT NULL,
  `a_sId` int NOT NULL,
  PRIMARY KEY (`aDate`),
  CONSTRAINT `a_sid` FOREIGN KEY (`a_sId`) REFERENCES `student` (`sId`)
)

CREATE TABLE `module` (
  `mId` int NOT NULL AUTO_INCREMENT,
  `mMath` int DEFAULT NULL,
  `mPhysics` int DEFAULT NULL,
  `mChemistry` int DEFAULT NULL,
  `mEnglish` int DEFAULT NULL,
  `mHistory` int DEFAULT NULL,
  `mBiology` int DEFAULT NULL,
  `mGeography` int DEFAULT NULL,
  `m_sId` int DEFAULT NULL,
  PRIMARY KEY (`mId`),
  CONSTRAINT `m_sId` FOREIGN KEY (`m_sId`) REFERENCES `student` (`sId`)
)
